
def pbMountCar
  return if $PokemonGlobal.car
  $PokemonGlobal.car = true
  pbUpdateVehicle
  bike_bgm = GameData::Metadata.get.bicycle_BGM
  pbCueBGM(bike_bgm, 0.5) if bike_bgm
  pbPokeRadarCancel
end

def pbDismountCar
  return if !$PokemonGlobal.car
  $PokemonGlobal.car = false
  pbUpdateVehicle
  $game_map.autoplayAsCue
end
